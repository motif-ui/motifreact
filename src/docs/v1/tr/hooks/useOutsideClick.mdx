# useOutsideClick

Belirtilen bir öğenin dışındaki tıklamaları algılar; açılır menüleri (dropdowns), modalları ve açılır pencereleri (popovers) kapatmak için kullanışlıdır.

## Kullanım

```tsx
import { useOutsideClick } from "@motif-ui/react";

const Dropdown = () => {
  const [isOpen, setIsOpen] = useState(true);
  const onOutsideClick = () => {
    setIsOpen(false);
  };
  const dropdownRef = useOutsideClick<HTMLDivElement>(onOutsideClick);

  return (
    <div ref={dropdownRef}>
      {isOpen && <div>Dropdown content</div>}
    </div>
  );
};
```

## Hariç Tutulan Ref'lerle

Belirli öğeleri dış tıklamanın tetiklenmesinden hariç tutabilirsiniz:

```tsx
import { useRef } from "react";
import { useOutsideClick } from "@motif-ui/react";

const Modal = () => {
  const buttonRef = useRef<HTMLButtonElement>(null);
  const modalRef = useOutsideClick<HTMLDivElement>(
    () => console.log("Dışarıya Tıklandı"),
    [buttonRef] // Düğme tıklamaları geri çağrıyı tetiklemez
  );

  return (
    <>
      <button ref={buttonRef}>Modali Aç</button>
      <div ref={modalRef}>Modal içeriği</div>
    </>
  );
};
```
