# useOutsideClick

Detects clicks outside of a specified element, useful for closing dropdowns, modals, and popovers.

## Usage

```tsx
import { useOutsideClick } from "@motif-ui/react";

const Dropdown = () => {
  const [isOpen, setIsOpen] = useState(true);
  const onOutsideClick = () => {
    setIsOpen(false);
  };
  const dropdownRef = useOutsideClick<HTMLDivElement>(onOutsideClick);

  return (
    <div ref={dropdownRef}>
      {isOpen && <div>Dropdown content</div>}
    </div>
  );
};
```

## With Excluded Refs

You can exclude specific elements from triggering the outside click:

```tsx
import { useRef } from "react";
import { useOutsideClick } from "@motif-ui/react";

const Modal = () => {
  const buttonRef = useRef<HTMLButtonElement>(null);
  const modalRef = useOutsideClick<HTMLDivElement>(
    () => console.log("Clicked outside"),
    [buttonRef] // Button clicks won't trigger the callback
  );

  return (
    <>
      <button ref={buttonRef}>Open Modal</button>
      <div ref={modalRef}>Modal content</div>
    </>
  );
};
```
